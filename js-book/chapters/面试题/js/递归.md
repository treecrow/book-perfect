# 递归

## 返回一个数的各个因子

```js
function getPreGens(a) {
  var arr = [];
  var n = Math.floor(a / 2) + 1;
  var t = n;
  var i;
  while (1 == 1) {
    if (i == t) {
      arr.push(a);
      break;
    }

    for (i = 2; i < t; i++) {
      if (a % i == 0) {
        arr.push(i);
        a = a / i;
        t = Math.floor(a / 2) + 1;
        break;
      }
    }
  }
  return arr;
}
alert(getPreGens(180)); //2,2,3,3,5
```
