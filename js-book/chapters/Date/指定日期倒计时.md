## 到指定日期时间的倒计时
```js
function getCountDown(Y, M, D, h, m, s){
    Y = Y || 0;
    M = M || 0;
    D = D || 0;
    h = h || 0;
    m = m || 0;
    s = s || 0;
    var date = new Date(Y, M-1, D, h, m, s),
    //转换为时间戳，方便计算差值
        times = date.getTime() - new Date().getTime();
    //返回天数
    return Math.ceil(times / (1000 * 60 * 60 * 24));
}
var tian=getCountDown("2016", "10", "1");//距离今年10月1日还有多少天
```
