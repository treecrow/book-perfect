### 判断变量是否空值
```js
function isEmpty(val){
  switch (typeof val) {
    case 'undefined' :
      return true;
      break;
    case 'string'    :
      if (val.trim().length == 0) return true;
      break;
    case 'boolean'   :
      if (!val) return true;
      break;
    case 'number'    :
      if (0 === val) return true;
      break;
    case 'object'    :
      if (null === val) return true;
      if (undefined !== val.length && val.length == 0) return true;
      for (var k in val) {
        return false;
      }
      return true;
      break;
  }
  return false;
}
```
