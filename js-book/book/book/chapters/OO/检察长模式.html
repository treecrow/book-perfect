检察长模式

检察长模式

避免在创建一个类的示例的时候，忘记写 new

示例

function Book(title, time, type) {
  // 判断执行过程中 this 是否是当前这个对象
  if (this instanceof Book) {
    this.title = title;
    this.time = time;
    this.type = type;
  // 否则重新创建这个对象
  } else {
    return new Book(title, time, type);
  }
}
