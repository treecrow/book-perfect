1.  解:

```js
function getArr(arr, sum) {
  // 首先给数组排序
  arr = arr.sort();
  // 用户存放返回结果
  const result = [];

  // 从数组的开始遍历到结束
  for (let i = 0; i < arr.length; i++) {
    let num = arr[i];
    // 遍历当前元素在数组中后面的元素
    for (let j = i + 1; j < arr.length; j++) {
      let num2 = arr[j];
      // 如果两个数之和等于 sum 则保存到返回结果里
      num + num2 === sum && result.push([num, num2]);
    }
  }
  // 返回
  return result;
}
// 打印
console.log(getArr([10, 5, 6, 7, 2, 8, 1, 9, 14], 15));
```

3.  解:转换问题:给定一个数（和），这个数有多个不大于 m 的数组成，共有多少种组合形式？

```js
function getMethodCount(m, sum) {
  return (sum - m) * m + ((m + 1) / 2) * m;
}
```
