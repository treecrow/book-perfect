使用示例

  • 更新 slick
if (!whole.localeSlick) {
  whole.localeSlick = $("#localeSlick").slick({
    autoplay: false,
    autoplaySpeed: 3000,
    // arrows: false,
    // appendArrows: '',
    // prevArrow: '',
    // nextArrow: '',
    infinite: false
  });
} else {
  whole.localeSlick.slick("setPosition");
}
  • 设置自定义箭头按钮
$(".slider").slick({
  appendArrows: $(".slider-container"),
  prevArrow: '<span class="prev">prev</span>',
  nextArrow: '<span class="next">next</span>',
  infinite: false,
  slidesToShow: 8,
  slidesToScroll: 8
});
