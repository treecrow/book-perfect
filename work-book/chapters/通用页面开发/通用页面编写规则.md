## url 参数的处理

```javascript
function getUrlParams() {
  var searchArr = window.location.search.substring(1).split("&");
  var obj = {};
  searchArr.forEach(function(item) {
    var arr = item.split("=");
    obj[decodeURI(arr[0])] = decodeURI(arr[1]);
  });
  return obj;
}
```
