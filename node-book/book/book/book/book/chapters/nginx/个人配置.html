user root owner;
worker_processes  1;
events {
  worker_connections  1024;
}


http {
  include       mime.types;
  default_type  application/octet-stream;
  sendfile        on;
  keepalive_timeout  65;

  # 本地
  server {
    listen       8080;
    server_name  localhost;
    location / {
        root   html;
        index  index.html index.htm;
    }
    error_page   500 502 503 504  /50x.html;
    location = /50x.html {
        root   html;
    }
  }

  # sassOperate（售后服务系统---运营端）
  server {
    listen       80;
    server_name  service.d.renrenauto.com;
    location / {
      proxy_pass http://10.4.82.23;
      proxy_set_header Host $host;
    }
    location ^~ /pages {
       root /Users/wangzhihui/Documents/rr-works/sassOperate-project;
       index index.html index.htm;
    }
    autoindex       on;
    autoindex_exact_size    on;
  }

    autoindex       on;
    autoindex_exact_size    on;

    rewrite ^/a?([0-9])+/(.*)$ /$2 last;
    rewrite (.*)\.[0-9]+\.css /$1.css;
    rewrite (.*)\.[0-9]+\.js /$1.js;

    location / {
      index   index.html index.htm;
    }

    location ~* \.(?:ttf|ttc|otf|eot|woff)$ {
      add_header "Access-Control-Allow-Origin" "*";
      expires 1M;
      access_log off;
      add_header Cache-Control "public";
