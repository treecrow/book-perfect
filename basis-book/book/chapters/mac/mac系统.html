安装 brew

/usr/bin/ruby -e "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install)"

安装 nvm 、node

  • nvm
// 安装NVM
brew install nvm
// 创建.bash_profile文件
touch ~/.bash_profile
// 向.bash_profile添加如下内容
export NVM_DIR=~/.nvm
source $(brew --prefix nvm)/nvm.sh
// 生效
source ～/.bash_profile
  • node && npm 
// 查看 Node 版本
nvm ls-remote node
// 选择Node 版本安装
nvm install v6.9.1
// 使用制定版本 Node
nvm use v5.0.0
nvm alias default v5.0.0

安装 zsh

  1. 安裝 zsh
// 安装
brew install zsh
// 将預設的 shell 修改为 zsh
chsh -s /bin/zsh
  1. 安裝 oh-my-zsh
git clone git://github.com/robbyrussell/oh-my-zsh.git ~/.oh-my-zsh
cp ~/.oh-my-zsh/templates/zshrc.zsh-template ~/.zshrc
  1. 修改 .zshrc
// 打开.zshrc
open -e ~/.zshrc
// 添加下面内容
source ~/.bash_profile

设置 github ssh

// 设置git的user.name和email
git config --global user.name "treecrow"
git config --global user.email "drytreecrow@gmail.com" 
// 生成私钥和公钥
ssh-keygen -t rsa -C "drytreecrow@gmail.com"
ssh-add -K id_rsa
// 添加 私密钥 到ssh(一般不需要做这一步)
id_rsa.pub
// 拷贝 id_rsa.pub 中的公钥到github

.DS_Store

禁止.DS_store生成

defaults write com.apple.desktopservices DSDontWriteNetworkStores -bool TRUE
// 重启Mac

恢复.DS_store生成

defaults delete com.apple.desktopservices DSDontWriteNetworkStores
